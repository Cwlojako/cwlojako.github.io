# 快速实现瀑布流布局

有左边固定，右边自适应的布局，也有左右固定，中间自适应的布局.....。布局千千万，布局的选择也依赖于网页的结构

那么瀑布流是什么？瀑布流，顾名思义就是像瀑布一样垂直的排列，有长有短，如下图所示： 显然，这时候我们采取flex布局来实现会非常麻烦，因为元素的长度取决于自身，flex布局无法实现精准分配。所以，就轮到瀑布流的登场啦，接下来代码需要用到jquery，方便获取dom元素以及操作。

需求知道了，我们来构建思路逻辑，这里我们假设每个元素宽度固定，大部分需求也是如此。关键思想是****（将第一行的元素高度存入一个数组中，找出最小值，第二行的第一个元素添加在第一行最小高度元素的下面，再更新数组里的元素高度）****

  1. 根据屏幕宽度或者父元素的宽度 与 元素宽度计算出每一行能容纳几个元素，也即是列数
  2. 把第一行的元素存入到数组中并进行定位
  3. 在数组中找到最小值即第一行中的最小高度
  4. 根据最小高度找到对应索引值
  5. 对其他行进行定位，并遵循以下规则 left: 最小高度索引 * 元素宽度 + ‘px’ top: 最小高度 + 'px'
  6. 高度数组更新叠加

<script type="text/javascript">
	// 获取元素
	let box = $('.box')
	// 定义列数
	let colCount = 0
	// 定义第一行高度数组
	let heightArr = []
	// 获取元素宽度
	// outerWidth(true) = xxx + padding + border + margin;
	let boxWidth = $('.box').outerWidth(true)
    // 调用瀑布流布局方法
	waterFall()

	// 窗口大小改变时自动触发
	$(window).on('resize',function(){
		waterFall()
	})

	// 窗口加载完成时
	$(window).on('load',function(){
		waterFall()
	})

	function waterFall(){
		$.each($('.box'),function(index,item){
			// 根据屏幕宽度求出列数
			colCount = parseInt($(window).width()/boxWidth)
			// outerWidth(true) = xxx + padding + border + margin;
			// 获取元素高度，包含外边距
			let boxHeight = $(item).outerHeight(true)
			if(index < colCount){
				//如果是第一行
				$(item).css({
					position: 'absolute',
					left: index*boxWidth+'px',
					top: 0
				})
				// 赋值高度数组
				heightArr[index] = boxHeight
			} else {
				console.log(heightArr)
				// 求出第一行中的最小高度
				let minBoxHeight = Math.min(...heightArr)
				// 根据最小高度找到最小高度元素的索引值
				let minBoxIndex = heightArr.findIndex(item => item == minBoxHeight)
				// 依照上面思路第五条的规则开始进行定位
				$(item).css({
					position: 'absolute',
					left: minBoxIndex*boxWidth+'px',
					top: minBoxHeight+'px'
				})

				// 关键一步，高度数组更新叠加
				heightArr[minBoxIndex] += boxHeight
			}
		})
	}
</script>
